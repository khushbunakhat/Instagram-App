class Main {
	public static void main (String[] args) {
       Scanner sc = new Scanner(System.in);
	  int q=sc.nextInt();
	  for(int i=0;i<q;i++){
		long n = sc.nextLong();
        long totalCost=0;
        long grp=0;
        killCandy(n,totalCost,grp);
         
      
   
    }
   
	}
     public static void killCandy(long n, long totalCost, long grp){
        if(n>0){
            grp = ((n-1)/3)+1;
            totalCost += grp*grp;
            killCandy(n-grp,totalCost,grp);
           
		}
        else{
             System.out.println(totalCost);
            return;
             
            
        }
     

     }
}
